Guardar en tabla ganadores (para el calendario interactivo) const { error: err1 } = await _supabase.from('ganadores').insert([ { evento_id: idT, nombre_jugador: nomG, premio: ptsNuevos + " PTS" } ]); // 2. Obtener puntos actuales del ranking para sumar const { data: userExistente } = await _supabase .from('ranking') .select('puntos') .eq('nombre', nomG) .single(); let puntosPrevios = userExistente ? Number(userExistente.puntos) : 0; let totalActualizado = puntosPrevios + ptsNuevos; // 3.
